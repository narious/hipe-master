19//ExpArray abstract data type
20//useful for arrays that need to be resized in O(1) time while preserving pointers to existing elements.
21//otherwise similar to ordinary arrays, including (theoretical) O(1) access time to known element index.
22
23//has size 1 initially.
24
25//note that elements are numbered from [1]. There is no [0].
26
27#pragma once
28
29#include <cstdlib>
30#include <stddef.h> //for size_t type
31
32
33template <class T> class ExpArray {
34private:
35 T** data; //data is an array of T arrays
36 short capacityMsb; //capacity of the most-significant-bit array
37 short sizeMsb; //most significant bit position in the currently allocated array size. E.g. sizeMsb==0 means 2^0==1
38
39 //returns the place of the most significant bit of the binary representation of num
40 //with the least significant bit represented as zero (2^0 = 1)
41 //if num==0, -1 is returned.
42 short msbOf(size_t num) {
43 short msbExponent = -1;
44 while(num) {
45 num >>= 1;
46 msbExponent++;
47 }
48 return msbExponent;
49 }
50public:
51 ExpArray() {
52 capacityMsb=0;
53 data = (T**) malloc((capacityMsb+1) * sizeof(T**));
54 sizeMsb=0; //initial size is 1==2^0
55 data[0] = new T[1](); //allocate a size 1 array to store the first element
56 }
57
58 ExpArray(const ExpArray&) {
59 //copy constructor to prevent dangerous copying.
60 ExpArray(); //just make a new blank array for now.
61 //TODO: make a proper copy.
62 }
63
64 ~ExpArray() {
65 while(sizeMsb>=0) shrink();
66 free(data);
67 }
68
69 size_t size() { //returns the number of elements available
70 return ((1<<sizeMsb)<<1) - 1; //subtract 1 because element [0] unused
71 }
72
73 size_t shrunkSize() { //returns what the size() will become if the array is shrunk with shrink()
74 return ((1<<(sizeMsb-1))<<1) - 1; //same form as size() with sizeMsb decremented
75 }
76
77 void grow() { //approximately double the array size
78 if(sizeMsb == capacityMsb) //need to grow the data array-of-arrays to add the next array
79 data = (T**) realloc(data, (++capacityMsb+1) * sizeof(T));
80 sizeMsb++;
81 data[sizeMsb] = new T[1<<sizeMsb]();
82 }
83
84 void shrink() { //approximately halve the array size
85 delete[] data[sizeMsb--];
86 }
87
88 //elements numbered from [1]. Element [0] makes no sense in the context of this data structure.
89 //the 'by reference' return allows this to be used for both accessing and assigning values.
90 T& operator[](size_t i) {
91 short msb = msbOf(i);
92 i &= ~(1<<msb); //remove most significant bit
93 return data[msb][i];
94 }
95
96};

99//AutoExpArray -- a user-friendlyness wrapper for ExpArray.
100//Purpose: Allow user to worry about number of elements used, rather than current allocated capacity.
101//Automatically increase the capacity if the user requests more elements than allocated, and shrink
102//automatically too to conserve memory.
103
104
105//New usage: Use inheritance rather than wrapping (save 4 bytes per instance).
106//Instead of having addElement and delElement functions, use setSize to resize the number of 'used'
107//elements dynamically. Behind the scenes, the array grows and shrinks in 2^n increments -- not
108//every setSize results in a resize behind the scenes. (So elements may or may not be freed following
109//a setSize call. At any rate, they should no longer be referenced.
110//- base class functions grow() or shrink(), these are called automatically,
111//and we insulate the user from these.
112
113
114
115template <class T> class AutoExpArray : protected ExpArray<T> {
116 private:
117 size_t used; //the number of elements in use
118 public:
119 AutoExpArray() {
120 used = 0;
121 }
122
123 size_t size() {
124 //use ExpArray::size() for base class meaning
125 return used;
126 }
127
128 void setSize(size_t newSize) {
129 if(newSize > used) {
130 while(newSize > ExpArray<T>::size())
131 ExpArray<T>::grow();
132 } else { //determine if shrinking is worthwhile based on amortised cost as discussed in Katajainen and Mortensen
133 if(newSize < (ExpArray<T>::size() >> 2))
134 ExpArray<T>::shrink();
135 }
136 used = newSize;
137 }
138
139 T& operator[](size_t i) {
140 return ExpArray<T>::operator[](i);
141 }
142};
