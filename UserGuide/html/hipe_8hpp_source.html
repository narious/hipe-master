132inline void session::incrementReferenceCount(hipe_loc location) {
133//increments our local reference count for the location
134 if(location == 0) return; //the body element is always 0, not requiring allocation.
135 try { //try to increment referenceCounts.at(location)
136 //the [] operator creates elements if they don't exist, while .at() throws exception.
137 referenceCounts[location] = referenceCounts.at(location) + 1;
138 } catch(const std::out_of_range& e) { //throws execption if reference nonexistent.
139 referenceCounts[location] = 1;
140 }
141}
142
143inline void session::decrementReferenceCount(hipe_loc location) {
144//decrmements local reference count for a particular location, or frees the location
145//if its reference count is zero.
146 if(location == 0) return; //the body element is always 0, not requiring allocation.
147 size_t currentCount = referenceCounts[location];
148 if(currentCount == 0) {
149 //free this location.
150 hipe_send((hipe_session) *this, HIPE_OP_FREE_LOCATION, 0, location, 0,0);
151 } else {
152 referenceCounts[location] = currentCount-1;
153 }
154}

205inline loc& loc::operator= (const loc& orig) {
206//copy assignment operator
207 if(&orig == this) return *this; //guard against self-assignment.
208 if(_session)

231inline loc loc::firstChild() {
232//return first child node of this element
233 hipe_send(*_session, HIPE_OP_GET_FIRST_CHILD, 0, location, 0, 0);
234 hipe_instruction instruction;
235 hipe_instruction_init(&instruction);
236 hipe_await_instruction(*_session, &instruction, HIPE_OP_LOCATION_RETURN);
237 hipe_loc location = instruction.location;
238 hipe_instruction_clear(&instruction);
239 return loc(location, _session);
240}

287inline int loc::send(char opcode, uint64_t requestor, const std::vector<std::string>& args) {
288//sends an instruction with this element passed as the location to act on.
289
290 int result;
291 hipe_instruction instruction;
292 hipe_instruction_init(&instruction);
293 instruction.opcode = opcode;
294 instruction.requestor = requestor;
295 instruction.location = location;
296 int i;
297 for(i=0; i<args.size(); i++) {
298 if(args[i].size()) {
299 instruction.arg[i] = (char*) args[i].data();
300 instruction.arg_length[i] = args[i].size();
301 }
302 }
303 result = hipe_send_instruction(*_session, instruction);
304 return result;
305}
