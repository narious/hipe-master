476/*Must be called on any new hipe_instruction instance (excepting shallow copies
477 *of an existing instance). The same applies if a struct previously modified by
478 *the user (e.g. assigning arg strings to it) is going to be reused for
479 *collecting new input from hipe. This function doesn't allocate or free memory,
480 *but it initialises the struct's values into a consistent state so that other
481 *hipe functions can do so safely.
482 *
483 * O(1) worst case time complexity.
484 */
485void hipe_instruction_init(hipe_instruction* obj);
486
487/*Allocates memory to store each argument in obj, according to arg_length[]
488 *values already assigned to obj. Arguments allocated in this way should later
489 *be freed with hipe_instruction_clear().
490 *
491 * O(n) worst case complexity where n is the length of an argument from mallocing
492 * arguments, but may as well be O(1) because the max length of a string is 64 bits.
493 */
494void hipe_instruction_alloc(hipe_instruction* obj);
495
496/*Frees memory previously allocated within the instruction, and leaves the
497 *struct in a clean state ready for reuse. Do not use this function to clear a
498 *hipe_instruction for which you have allocated values yourself. Instead,
499 *de-allocate the argument strings in a way consistent with how you allocated
500 *them, then call hipe_instruction_init() to re-initialise the values into a
501 *clean state.
502 *
503 * O(1) worst case time complexity.
504 */
505void hipe_instruction_clear(hipe_instruction* obj);
506
507/*Deeply copies src values into dest, allocating memory for args.
508 *Does not do any initialisation or deallocation of any previous values that may
509 *have been present in dest previously.
510 *When done with the instruction in dest, free the memory of its internal
511 *variables with hipe_instruction_clear().
512 *
513 * See hipe_instruction_alloc for complexity.
514 */
515void hipe_instruction_copy(hipe_instruction* dest, hipe_instruction* src);
516
517/*Moves allocations from src to dest. Similar to hipe_instruction_copy except no
518 *new memory is allocated; the memory allocated to src is reassigned to dest,
519 *and src is then re-initialised. The memory in dest should later be freed in a
520 *manner consistent with how src was previously allocated.
521 *
522 * O(1) worst case time complexity.
523 */
524void hipe_instruction_move(hipe_instruction* dest, hipe_instruction* src);
